<script>
    function copyResult() {
        if(!document.getElementById('agree').checked) {
            alert('안내사항 동의에 체크해주세요!'); 
            return; 
        }

        var getVal = function(id) { return document.getElementById(id).value || '미입력'; };
        var getCheck = function(name) {
            var checked = Array.from(document.querySelectorAll('input[name="' + name + '"]:checked'));
            return checked.map(function(el) { return el.value; }).join(', ') || '선택안함';
        };

        var text = "[DAON 신청서]\n" +
                   "■ 채널: " + getVal('channel') + "\n" +
                   "■ 원곡: " + getVal('original_link') + "\n" +
                   "■ 용도: " + getVal('usage') + "\n" +
                   "■ 타입: " + getCheck('type') + "\n" +
                   "■ 언어: " + getCheck('lang') + "\n" +
                   "■ 옵션: " + getCheck('opt') + "\n" +
                   "■ 마감일: " + getVal('deadline') + "\n" +
                   "■ 연출/크레딧: \n" + getVal('credit');

        var t = document.createElement("textarea");
        document.body.appendChild(t);
        t.value = text;
        t.select();
        document.execCommand('copy');
        document.body.removeChild(t);
        
        alert('양식이 복사되었습니다! 아트머그 문의창에 붙여넣어 주세요. ※내용 확인 없이 체크박스에 확인하신 경우 생기는 문제에 대해 책임은 의뢰주에게 있습니다.');
    }
</script>
